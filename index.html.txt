  <h1>LeadCar CRM</h1>

  <input id="nome" placeholder="Nome do cliente">

  <select id="status">
    <option>Novo</option>

    <option>Perdido</option>
  </select>

  <button onclick="adicionarLead()">Adicionar Lead</button>

  <h2>Leads</h2>
  <div id="lista"></div>

  <script>
    let leads = JSON.parse(localStorage.getItem("leads")) || [];

    function salvar() {
      localStorage.setItem("leads", JSON.stringify(leads));
    }

    function adicionarLead() {
      const nome = document.getElementById("nome").value;
      const telefone = document.getElementById("telefone").value;
      const carro = document.getElementById("carro").value;
      const status = document.getElementById("status").value;

      if (!nome || !telefone || !carro) {
        alert("Preencha nome, telefone e carro.");
        return;
      }

      leads.push({
        nome,
        telefone,
        carro,
        status,
        ultimoContato: new Date().toLocaleDateString("pt-BR")
      });

      salvar();
      renderizar();

      document.getElementById("nome").value = "";
      document.getElementById("telefone").value = "";
      document.getElementById("carro").value = "";
    }

    function gerarMensagem(lead) {
      return `Oi ${lead.nome}, tudo bem? Vi que você demonstrou interesse no ${lead.carro}. Ainda está procurando esse modelo ou posso te mostrar algumas opções parecidas?`;
    }

    function abrirWhatsApp(telefone, mensagem) {
      const numero = telefone.replace(/\D/g, "");
      const texto = encodeURIComponent(mensagem);
      window.open(`https://wa.me/55${numero}?text=${texto}`, "_blank");
    }

    function excluirLead(index) {
      leads.splice(index, 1);
      salvar();
      renderizar();
    }

    function renderizar() {
      const lista = document.getElementById("lista");
      lista.innerHTML = "";

      leads.forEach((lead, index) => {
        const classe = lead.status === "Negociando" ? "hot" : lead.status === "Perdido" ? "cold" : "";

        lista.innerHTML += `
          <div class="lead ${classe}">
            <h3>${lead.nome}</h3>
            <p><strong>WhatsApp:</strong> ${lead.telefone}</p>
            <p><strong>Carro:</strong> ${lead.carro}</p>
            <p><strong>Status:</strong> ${lead.status}</p>
            <p><strong>Último contato:</strong> ${lead.ultimoContato}</p>

            <button onclick='abrirWhatsApp("${lead.telefone}", "${gerarMensagem(lead)}")'>
              Chamar no WhatsApp
            </button>

            <button onclick="excluirLead(${index})">
              Excluir
            </button>
          </div>
        `;
      });
    }

    renderizar();
  </script>
